          # 适用于容器的 Windows

          # 运行环境

           Windows Server 2016及以上 / Windows 10

          # 安装Docker

          # Windows Server

          1. 打开提升的 PowerShell 会话，从 PowerShell 库安装 Docker-Microsoft PackageManagement 提供程序。 Install-Module -Name DockerMsftProvider -Repository PSGallery -Force
          2. 使用 PackageManagement PowerShell 模块安装最新版本的 Docker。 Install-Package -Name docker -ProviderName DockerMsftProvider
          3. 在安装完成后，请重启计算机。 Restart-Computer -Force

          # Windows 10

          1. 下载安装 Docker Desktop
          2. 将默认容器类型设置为 Windows 容器 ？ & $Env:ProgramFiles\Docker\Docker\DockerCli.exe -SwitchDaemon .

          # 编写Dockerfile

          FROM mcr.microsoft.com/dotnet/aspnet:3.1 AS base
WORKDIR /app
EXPOSE 12041

FROM mcr.microsoft.com/dotnet/sdk:3.1 AS build
WORKDIR /src

COPY ./NuGet.Config .
COPY ./APos.Pay.Interface/APos.Pay.Interface.csproj ./APos.Pay.Interface/
COPY ./APos.Pay.Application/APos.Pay.Application.csproj ./APos.Pay.Application/
COPY ./APos.Pay.IApplication/APos.Pay.IApplication.csproj ./APos.Pay.IApplication/
COPY ./APos.Pay.DataContract/APos.Pay.DataContract.csproj ./APos.Pay.DataContract/
COPY ./APos.Pay.Domain/APos.Pay.Domain.csproj ./APos.Pay.Domain/
COPY ./APos.Pay.DataAccess/APos.Pay.DataAccess.csproj ./APos.Pay.DataAccess/

#恢复项目的依赖项和工具
RUN dotnet restore "APos.Pay.Interface/APos.Pay.Interface.csproj"
COPY . .
WORKDIR "/src/APos.Pay.Interface"
#生成项目及其所有依赖项
RUN dotnet build "APos.Pay.Interface.csproj" -c Release -o /app/build

FROM build AS publish
#发布
RUN dotnet publish "APos.Pay.Interface.csproj" -c Release -o /app/publish

FROM base AS final
WORKDIR /app
COPY --from=publish /app/publish .
ENTRYPOINT ["dotnet", "APos.Pay.Interface.dll"]
